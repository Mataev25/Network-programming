cmake_minimum_required (VERSION 3.8)
project (NetworkProject)
set(Boost_USE_STATIC_LIBS OFF)
set(Boost_USE_DEBUG_LIBS ON)  
set(Boost_USE_MULTITHREADED ON)



find_package(Boost 1.87 REQUIRED COMPONENTS thread filesystem system)
if(Boost_FOUND)
    message(STATUS "Boost version: ${Boost_VERSION}")
    message(STATUS "Boost include dirs: ${Boost_INCLUDE_DIRS}")
    message(STATUS "Boost libraries: ${Boost_LIBRARIES}")
endif()

add_executable ( Client "Client.cpp" "Client.h")

target_link_libraries(Client PRIVATE 
    Boost::thread 
    Boost::filesystem 
    Boost::chrono
    Boost::system
)

add_executable(Server2.0 "Server2.0")
target_link_libraries(Server2.0 PRIVATE 
    Boost::system
)


if (MSVC)
    target_compile_options(Client PRIVATE /std:c++17 /permissive-)
    target_compile_options(Server2.0 PRIVATE /std:c++17 /permissive-)
    
else()
    target_compile_options(Client PRIVATE -std=c++17)
    target_compile_options(Server2.0 PRIVATE -std=c++17)
    
endif()
